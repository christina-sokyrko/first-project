git init - створити репозиторій

git status - перевірити, що відбувається в репозиторії

git add - додати невідстлідковуваний файл

git commit - створити версію проекту

git log - подивитися всі коміти і їх деталі

git checkout - перейти до будь-якої версії

git branch - керування гілками проекту

git push - вивантажити всю історію у віддалений репозиторій

git pull - стягнути зміни з віддаленого репозиторія в локальний

git merge - злити гілки в одну

git clone - "клонувати" репозиторій в нову директорію
